let transactions = JSON.parse(localStorage.getItem('txns') || '[]');

const fmt = n => n.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
const fmtDate = d => { try { return new Date(d).toLocaleDateString('en-US',{month:'short',day:'numeric'}); } catch(e){return d;} };
const save = () => {
  localStorage.setItem('txns', JSON.stringify(transactions));

function showScreen(name) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
  document.getElementById('screen-' + name).classList.add('active');
  const nav = document.getElementById('nav-' + name);
  if (nav) nav.classList.add('active');
  currentScreen = name;
  if (name === 'home')          renderHome();
  if (name === 'transactions')  renderAll();
  if (name === 'subscriptions') renderSubs();
  if (name === 'analytics')     renderAnalytics();
  window.scrollTo(0, 0);
}

/* ════════════════════════════════════════════════════
   RENDER HOME
════════════════════════════════════════════════════ */
function renderHome() {
  const income   = transactions.filter(t=>t.type==='income').reduce((s,t)=>s+t.amount,0);
  const expenses = transactions.filter(t=>t.type==='expense').reduce((s,t)=>s+t.amount,0);
  const balance  = income - expenses;

  animateCount('home-balance', balance, true);
  document.getElementById('home-income').textContent   = '+$' + fmt(income);
  document.getElementById('home-expenses').textContent = '-$' + fmt(expenses);

  const subTotal = subscriptions.reduce((s,x)=>s+x.amount,0);
  document.getElementById('home-sub-total').textContent = '$' + fmt(subTotal) + '/mo';
  document.getElementById('home-sub-info').textContent  = subscriptions.length
    ? `${subscriptions.length} subscription${subscriptions.length>1?'s':''} active`
    : '0 subscriptions · Add your first';

  const list  = document.getElementById('home-txn-list');
  const empty = document.getElementById('home-empty');
  const recent = [...transactions].reverse().slice(0, 5);

  // remove old rows
  list.querySelectorAll('.txn-row').forEach(r=>r.remove());
  empty.style.display = recent.length ? 'none' : 'block';

  recent.forEach((t, i) => {
    const row = buildTxnRow(t);
    row.style.animationDelay = (i * 0.07) + 's';
    list.appendChild(row);
  });
}

function renderAll() {
  const list  = document.getElementById('all-txn-list');
  const empty = document.getElementById('all-empty');
  list.querySelectorAll('.txn-row').forEach(r=>r.remove());

  const filtered = currentFilter === 'all'
    ? [...transactions].reverse()

  empty.style.display = filtered.length ? 'none' : 'block';
  filtered.forEach((t, i) => {
    const row = buildTxnRow(t);
    row.style.animationDelay = (i * 0.05) + 's';
    list.appendChild(row);
  });
}

/* ════════════════════════════════════════════════════
   RENDER ANALYTICS
════════════════════════════════════════════════════ */
function renderAnalytics() {
  const income   = transactions.filter(t=>t.type==='income').reduce((s,t)=>s+t.amount,0);
  const expenses = transactions.filter(t=>t.type==='expense').reduce((s,t)=>s+t.amount,0);
  const balance  = income - expenses;

  document.getElementById('a-income').textContent   = '$' + fmt(income);
  document.getElementById('a-expenses').textContent = '$' + fmt(expenses);

  const savingsPct = income > 0 ? Math.max(0, Math.round((balance/income)*100)) : 0;
  document.getElementById('a-savings-pct').textContent = savingsPct + '%';
  setTimeout(() => {
    document.getElementById('a-savings-bar').style.width = savingsPct + '%';
  }, 100);

  // Category breakdown
  const cats = ['food','transport','bills','shopping','subscription','other'];
  const catList = document.getElementById('a-cat-list');
  catList.innerHTML = '';
  if (expenses === 0) {
    catList.innerHTML = '<div style="color:var(--muted);font-size:13px;text-align:center;padding:20px 0">Add expense transactions to see breakdowns</div>';
  } else {
    cats.forEach(cat => {
      const total = transactions.filter(t=>t.type==='expense'&&t.category===cat).reduce((s,t)=>s+t.amount,0);
      if (!total) return;
      const pct = Math.round((total/expenses)*100);
      const el = document.createElement('div');
      el.className = 'progress-row';
      el.innerHTML = `
        <div class="progress-header">
          <span style="font-weight:600;text-transform:capitalize">${cat}</span>
          <span style="color:var(--muted)">$${fmt(total)}</span>
        </div>
        <div class="progress-track">
          <div class="progress-fill" style="width:0%" data-target="${pct}"></div>
        </div>
      `;
      catList.appendChild(el);
    });
    setTimeout(() => {
      catList.querySelectorAll('.progress-fill[data-target]').forEach(bar => {
        bar.style.width = bar.dataset.target + '%';
      });
    }, 100);
  }

  // Bar chart — last 6 months mock + current
  const months = ['Nov','Dec','Jan','Feb','Mar','Apr'];
  const mockData = [0, 0, 0, 0, 0, expenses];
  const maxVal = Math.max(...mockData, 1);
  const chart = document.getElementById('bar-chart');
  chart.innerHTML = '';
  months.forEach((m, i) => {
    const h = Math.round((mockData[i] / maxVal) * 70);
    const isNow = i === months.length - 1;
    const col = document.createElement('div');
    col.className = 'bar-col';
    col.innerHTML = `
      <div class="bar-fill ${isNow?'this-month':''}" style="height:${h || 4}px"></div>
      <span class="bar-lbl">${m}</span>
    `;
    chart.appendChild(col);
  });
}

function saveTransaction() {
  const label  = document.getElementById('f-label').value.trim();
  const amount = parseFloat(document.getElementById('f-amount').value);
  const date   = document.getElementById('f-date').value || new Date().toISOString().slice(0,10);

  if (!label)  { shake('f-label');  return; }
  if (!amount) { shake('f-amount'); return; }

  const icons = { food:'🍔', transport:'🚗', bills:'⚡', shopping:'🛍️', subscription:'🔁', income:'💼', other:'📌' };
  transactions.push({
    id: Date.now(), label, amount, type: currentType,
    category: currentType==='income' ? 'income' : currentCat,
    date, icon: icons[currentType==='income'?'income':currentCat], auto: false,
  });
  save();

  // reset
  document.getElementById('f-label').value  = '';
  document.getElementById('f-amount').value = '';
  setType('expense');
  setCat('food', document.querySelector('.cat-chip'));

  showScreen('home');
}

function shake(id) {
  const el = document.getElementById(id);
  el.style.animation = 'none';
  el.offsetHeight; // reflow
  el.style.animation = 'shake .4s ease';
  el.focus();
}

function openAddSub() {
  document.getElementById('sub-modal').style.display = 'flex';
  const today = new Date();
  today.setMonth(today.getMonth() + 1);
  document.getElementById('sub-date').value = today.toISOString().slice(0,10);
}

function saveSub() {
  const name   = document.getElementById('sub-name').value.trim();
  const cost   = parseFloat(document.getElementById('sub-cost').value);
  const emoji  = document.getElementById('sub-emoji').value.trim() || '📦';
  const date   = document.getElementById('sub-date').value;

  if (!name || !cost) return;

  subscriptions.push({ id: Date.now(), name, amount: cost, icon: emoji, next: fmtDate(date) });
  save();

  document.getElementById('sub-name').value  = '';
  document.getElementById('sub-cost').value  = '';
  document.getElementById('sub-emoji').value = '';
  document.getElementById('sub-modal').style.display = 'none';
  renderSubs();
}

function deleteSub(id) {
  subscriptions = subscriptions.filter(s => s.id !== id);
  save();
  renderSubs();
}

      <div class="wallet-notif-sub">$${fmt(txn.amount)} auto-tracked</div>
    </div>
    <button class="wallet-notif-close" onclick="this.parentElement.remove()">×</button>
  `;
  document.getElementById('notif-container').appendChild(el);
  requestAnimationFrame(() => { requestAnimationFrame(() => { el.classList.add('show'); }); });
  setTimeout(() => {
    el.classList.remove('show');
    setTimeout(() => el.remove(), 500);
  }, 4000);
}

/* ════════════════════════════════════════════════════
   ANIMATED COUNTER
════════════════════════════════════════════════════ */
function animateCount(id, target, currency=false) {
  const el = document.getElementById(id);
  const start = 0, dur = 800;
  const startTime = performance.now();
  function step(now) {
    const p = Math.min((now - startTime) / dur, 1);
    const ease = 1 - Math.pow(1 - p, 3);
    const val = start + (target - start) * ease;
    el.textContent = (currency ? '$' : '') + fmt(Math.abs(val));
    if (p < 1) requestAnimationFrame(step);
  }
  requestAnimationFrame(step);
}
